\begin{definition}[Closed Function]\label{def:closed-function}
	A function \(f:\bbR^{n}\rightarrow\bbR\) is said to be closed, if for each \(\alpha\in\bbR\), the sublevel set
	\begin{equation}
		\{\bfx\in\operatorname{dom}f\mid f(\bfx)\leq\alpha\},
	\end{equation}
	is closed.
\end{definition}
